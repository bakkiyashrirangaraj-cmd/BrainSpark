
from fastapi import FastAPI, HTTPException, Depends
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
from typing import List, Optional
import httpx
import os

app = FastAPI(title="BrainSpark API", version="1.0.0")

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

ANTHROPIC_API_KEY = os.getenv("ANTHROPIC_API_KEY", "")
ANTHROPIC_API_URL = "https://api.anthropic.com/v1/messages"

class ChatRequest(BaseModel):
    message: str
    topic: str
    age_group: str
    history: Optional[List[dict]] = []

class ChatResponse(BaseModel):
    response: str
    stars_earned: int = 5

SYSTEM_PROMPTS = {
    "cubs": "You are BrainSpark, a friendly AI for children ages 4-6. Use VERY simple words, short sentences, lots of emojis. Be magical, fun, and encouraging! Always end with a simple question.",
    "explorers": "You are BrainSpark, an exciting AI for children ages 7-10. Use adventure metaphors and exciting examples. Be curious and encouraging! Always end with a thought-provoking question.",
    "masters": "You are BrainSpark, an intelligent AI for children ages 11-14. Engage with deeper questions and philosophical concepts. Challenge them intellectually while staying supportive! Always end with a deep question."
}

@app.get("/")
def root():
    return {"message": "BrainSpark API", "status": "running"}

@app.get("/health")
def health():
    return {"status": "healthy"}

@app.post("/api/chat", response_model=ChatResponse)
async def chat(request: ChatRequest):
    system_prompt = SYSTEM_PROMPTS.get(request.age_group, SYSTEM_PROMPTS["explorers"])
    system_prompt += f"\n\nCurrent topic: {request.topic}. Keep responses concise (2-3 paragraphs max)."
    
    messages = [{"role": m["role"], "content": m["text"]} for m in request.history[-10:]]
    messages.append({"role": "user", "content": request.message})
    
    try:
        async with httpx.AsyncClient() as client:
            response = await client.post(
                ANTHROPIC_API_URL,
                headers={
                    "Content-Type": "application/json",
                    "x-api-key": ANTHROPIC_API_KEY,
                    "anthropic-version": "2023-06-01"
                },
                json={
                    "model": "claude-sonnet-4-20250514",
                    "max_tokens": 800,
                    "system": system_prompt,
                    "messages": messages
                },
                timeout=30.0
            )
            data = response.json()
            ai_response = data.get("content", [{}])[0].get("text", "That's a great question!")
            return ChatResponse(response=ai_response, stars_earned=5)
    except Exception as e:
        return ChatResponse(response="That's fascinating! Tell me more about what you're thinking! ğŸ¤”", stars_earned=5)

-- BrainSpark Database Schema

CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    name VARCHAR(255),
    role VARCHAR(50) DEFAULT 'parent',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE children (
    id SERIAL PRIMARY KEY,
    parent_id INTEGER REFERENCES users(id),
    name VARCHAR(255) NOT NULL,
    age_group VARCHAR(50) NOT NULL,
    avatar VARCHAR(50) DEFAULT 'ğŸ§’',
    stars INTEGER DEFAULT 0,
    streak INTEGER DEFAULT 1,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE topics (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    emoji VARCHAR(10),
    unlock_at INTEGER DEFAULT 0
);

CREATE TABLE progress (
    id SERIAL PRIMARY KEY,
    child_id INTEGER REFERENCES children(id),
    topic_id INTEGER REFERENCES topics(id),
    questions_asked INTEGER DEFAULT 0,
    max_depth INTEGER DEFAULT 0,
    last_activity TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE conversations (
    id SERIAL PRIMARY KEY,
    child_id INTEGER REFERENCES children(id),
    topic_id INTEGER REFERENCES topics(id),
    messages JSONB DEFAULT '[]',
    depth INTEGER DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE achievements (
    id SERIAL PRIMARY KEY,
    child_id INTEGER REFERENCES children(id),
    achievement_key VARCHAR(100) NOT NULL,
    unlocked_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
